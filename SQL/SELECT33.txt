SELECT T2.ID FROM ECOLI_DATA E
LEFT JOIN ECOLI_DATA T1 ON T1.PARENT_ID = E.ID
LEFT JOIN ECOLI_DATA T2 ON T2.PARENT_ID = T1.ID
WHERE E.PARENT_ID IS NULL AND T1.PARENT_ID IS NOT NULL AND T2.PARENT_ID IS NOT NULL
ORDER BY 1
